        CSS
        <style>
            #btn-default{
                cursor: default;
            }
            #btn-pointer{
                cursor: pointer;
            }
            #btn-wait{
                cursor: wait;
            }
            #btn-no-drop{
                cursor: no-drop;
            }
        </style>
    
    ------------------------------------------------------------------------------------
    
    HTML
    <input id="btn-default" type="button" value="default">
    <input id="btn-pointer" type="button" value="pointer">
    <input id="btn-wait" type="button" value="wait">
    <input id="btn-no-drop" type="button" value="no-drop">
